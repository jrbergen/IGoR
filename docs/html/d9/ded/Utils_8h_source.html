103 template<typename T> struct Matrix {
104 public:
105  Matrix():rows(0) , cols(0) , array_p(new T [0]){}
106  Matrix(int m ,int n ): rows(m) , cols(n) , array_p(nullptr){
107  if(m*n>0 and m>0){
108  array_p = new T [m*n];
109  }
110  }
111  Matrix(int m , int n , T arr[]):rows(m) , cols(n) , array_p(new T [m*n]){
112  for(size_t i = 0 ; i != m*n ; i++){
113  array_p[i] = arr[i];
114  }
115  }
116  Matrix(int m , int n , std::vector<T> vect):rows(m) , cols(n) , array_p(new T [m*n]){
117  for(size_t i = 0 ; i != m*n ; i++){
118  array_p[i] = vect.at(i);
119  }
120  }
121  Matrix(const Matrix<T>& other){
122  //Provides deep copy of a matrix
123  this->rows = other.rows;
124  this->cols = other.cols;
125  this->array_p = new T [rows*cols];
126  for(int i = 0 ; i != rows*cols ; i++){
127  this->array_p[i] = other.array_p[i];
128  }
129  }
130  ~Matrix(){
131  delete [] array_p;
132  }
133 
134  Matrix<T>& operator=(const Matrix& other){
135  delete [] array_p;
136  this->rows = other.rows;
137  this->cols = other.cols;
138  this->array_p = new T [rows*cols];
139  for(int i = 0 ; i != rows*cols ; i++){
140  this->array_p[i] = other.array_p[i];
141  }
142  }
143 
144  T& operator()(const int& i ,const int& j ){
145  if( (i>rows-1) || (j>cols-1) ){
146  throw std::length_error("Cannot access indices ["+std::to_string(i)+","+std::to_string(j)+"] with matrix dimensions ["+std::to_string(rows)+","+std::to_string(cols)+"]");
147  std::cout<<"out_of range matrix coordinates: "<<rows<<"<"<<i<<" or "<<cols<<"<"<<j<<std::endl;
148  }
149  return array_p[i + rows*j];
150  }
151 
152  const T& operator()(const int& i ,const int& j ) const{
153  if( (i>rows-1) || (j>cols-1) ){
154  throw std::length_error("Cannot access indices ["+std::to_string(i)+","+std::to_string(j)+"] with matrix dimensions ["+std::to_string(rows)+","+std::to_string(cols)+"]");
155  std::cout<<"out_of range matrix coordinates: "<<rows<<"<"<<i<<" or "<<cols<<"<"<<j<<std::endl;
156  }
157  return array_p[i + rows*j];
158  }

160  T get_field(const int& i ,const int& j ) const{

162  throw std::length_error("Cannot access indices ["+std::to_string(i)+","+std::to_string(j)+"] with matrix dimensions ["+std::to_string(rows)+","+std::to_string(cols)+"]");
163  std::cout<<"out_of range matrix coordinates: "<<rows<<"<"<<i<<" or "<<cols<<"<"<<j<<std::endl;
164  }
165  return array_p[i + rows*j];
166  }
